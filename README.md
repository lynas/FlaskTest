# APP_SETTINGS

APP_SETTINGS is a environment variable stored in bashrc / zshrc

export APP_SETTINGS=/Users/sazzad/.usrConfig/app.cfg

app.cfg contains

MONGO_HOST='192.168.0.1'
MONGO_PORT=27017
MONGO_DBNAME='dbname'
MONGO_USERNAME='db_username'
MONGO_PASSWORD='db_password'
